        x = self.transition2(x)

        dep_x = self.block3(x)
        dia_x = self.dilationblock3(x)
        x = dep_x + dia_x
        x = self.out_block(x)

        x = x.view(-1, 10)
        return F.log_softmax(x, dim=-1)
